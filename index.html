Drupal → WordPress Migration · Quick look

This function walks nested Drupal components, preserves editor order, and filters out unpublished/missing nodes. Depth limit prevents runaway recursion.

// Depth-limited recursion, order preserved, invalid nodes skipped
function get_components(int $nid, int $depth = 0): array {
    if ($depth > 2) return [];
    $query = /* brand-specific SQL via SELECTED_SITE_ID */ $components_query;
    $rows = $this->drupalConnection->do_query($query, [$nid]);
    if (empty($rows[0]->field_components_target_id)) return [];

    $components = [];
    foreach ($rows as $row) {
        $id = (int)$row->field_components_target_id;
        $meta = $this->drupalConnection->do_query("SELECT type, status FROM node_field_data WHERE nid=%d;", [$id]);
        if (empty($meta[0]->type) || empty($meta[0]->status)) {
            // orphan/disabled → skip, optionally log
            continue;
        }
        $components[] = array_merge(
            $this->get_fields($id, $meta[0]->type),
            [
                'nid' => $id,
                'type' => $meta[0]->type,
                'status' => (bool)$meta[0]->status,
                'children' => $this->get_components($id, $depth + 1),
            ]
        );
    }
    return $components;
}

Programs Carousel (Gutenberg, SSR) · Quick look

Geo‑aware, server‑rendered carousel with taxonomy merging and clean integration hooks.

// Cookie → query → results (neutral hooks)
$cookie_key = apply_filters('programs_carousel/location_cookie_key', 'location_data');
$loc = isset($_COOKIE[$cookie_key]) ? json_decode(stripslashes($_COOKIE[$cookie_key]), true) : null;

$params = [
  'lat' => $loc['latitude'] ?? '',
  'lon' => $loc['longitude'] ?? '',
  'postal_code' => $loc['zipcode'] ?? '',
  'distance' => (int)($attributes['distance'] ?? 0),
  'listing_type' => sanitize_text_field($attributes['listingType'] ?? ''),
  'per_page' => 12,
];

$cats = $attributes['selectedCategories'] ?? [];
if (!empty($attributes['categoryParentPageInclude'])) {
  $defaultIds = array_map(fn($c) => $c->term_id, get_the_category(get_the_ID()) ?: []);
  $cats = array_unique(array_merge($defaultIds, $cats));
}

$params = apply_filters('programs_carousel/augment_query_with_categories', $params, $cats);
$programs = apply_filters('programs_carousel/fetch_programs', $params);
